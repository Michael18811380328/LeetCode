    еӯҗеәҸеҲ—й—®йўҳжҳҜеёёи§Ғзҡ„з®—жі•й—®йўҳпјҢиҖҢдё”е№¶дёҚеҘҪи§ЈеҶігҖӮ

    йҰ–е…ҲпјҢеӯҗеәҸеҲ—й—®йўҳжң¬иә«е°ұзӣёеҜ№еӯҗдёІгҖҒеӯҗж•°з»„жӣҙеӣ°йҡҫдёҖдәӣпјҢеӣ дёәеүҚиҖ…жҳҜдёҚиҝһз»ӯзҡ„еәҸеҲ—пјҢиҖҢеҗҺдёӨиҖ…жҳҜиҝһз»ӯзҡ„пјҢе°ұз®—з©·дёҫдҪ йғҪдёҚдёҖе®ҡдјҡпјҢжӣҙеҲ«иҜҙжұӮи§Јзӣёе…ізҡ„з®—жі•й—®йўҳдәҶгҖӮ

    д№ӢеүҚи§ЈеҶідәҶгҖҢжңҖй•ҝеӣһж–ҮеӯҗдёІгҖҚзҡ„й—®йўҳпјҢиҝҷж¬ЎжҸҗеҚҮйҡҫеәҰпјҢжұӮжңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—зҡ„й•ҝеәҰпјҡ

    жҲ‘д»¬иҜҙиҝҷдёӘй—®йўҳеҜ№ dp ж•°з»„зҡ„е®ҡд№үжҳҜпјҡеңЁеӯҗдёІ s[i..j] дёӯпјҢжңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—зҡ„й•ҝеәҰдёә dp[i][j]гҖӮдёҖе®ҡиҰҒи®°дҪҸиҝҷдёӘе®ҡд№үжүҚиғҪзҗҶи§Јз®—жі•гҖӮ

    дёәе•ҘиҝҷдёӘй—®йўҳиҰҒиҝҷж ·е®ҡд№үдәҢз»ҙзҡ„ dp ж•°з»„е‘ўпјҹжҲ‘д»¬еүҚж–ҮеӨҡж¬ЎжҸҗеҲ°пјҢжүҫзҠ¶жҖҒиҪ¬з§»йңҖиҰҒеҪ’зәіжҖқз»ҙпјҢиҜҙзҷҪдәҶе°ұжҳҜеҰӮдҪ•д»Һе·ІзҹҘзҡ„з»“жһңжҺЁеҮәжңӘзҹҘзҡ„йғЁеҲҶпјҢиҝҷж ·е®ҡд№үе®№жҳ“еҪ’зәіпјҢе®№жҳ“еҸ‘зҺ°зҠ¶жҖҒиҪ¬з§»е…ізі»гҖӮ

    е…·дҪ“жқҘиҜҙпјҢеҰӮжһңжҲ‘д»¬жғіжұӮ dp[i][j]пјҢеҒҮи®ҫдҪ зҹҘйҒ“дәҶеӯҗй—®йўҳ dp[i+1][j-1] зҡ„з»“жһңпјҲs[i+1..j-1] дёӯжңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—зҡ„й•ҝеәҰпјүпјҢдҪ жҳҜеҗҰиғҪжғіеҠһжі•з®—еҮә dp[i][j] зҡ„еҖјпјҲs[i..j] дёӯпјҢжңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—зҡ„й•ҝеәҰпјүе‘ўпјҹ

    еҸҜд»ҘпјҒиҝҷеҸ–еҶідәҺ s[i] е’Ң s[j] зҡ„еӯ—з¬Ұпјҡ

    еҰӮжһңе®ғдҝ©зӣёзӯүпјҢйӮЈд№Ҳе®ғдҝ©еҠ дёҠ s[i+1..j-1] дёӯзҡ„жңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—е°ұжҳҜ s[i..j] зҡ„жңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—пјҡ

    еҰӮжһңе®ғдҝ©дёҚзӣёзӯүпјҢиҜҙжҳҺе®ғдҝ©**дёҚеҸҜиғҪеҗҢж—¶**еҮәзҺ°еңЁ s[i..j] зҡ„жңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—дёӯпјҢйӮЈд№ҲжҠҠе®ғдҝ©**еҲҶеҲ«**еҠ е…Ҙ s[i+1..j-1] дёӯпјҢзңӢзңӢе“ӘдёӘеӯҗдёІдә§з”ҹзҡ„еӣһж–ҮеӯҗеәҸеҲ—жӣҙй•ҝеҚіеҸҜпјҡ

    д»ҘдёҠдёӨз§Қжғ…еҶөеҶҷжҲҗд»Јз Ғе°ұжҳҜиҝҷж ·пјҡ

    if (s[i] == s[j])
    // е®ғдҝ©дёҖе®ҡеңЁжңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—дёӯ
    dp[i][j] = dp[i + 1][j - 1] + 2;
else
    // s[i+1..j] е’Ң s[i..j-1] и°Ғзҡ„еӣһж–ҮеӯҗеәҸеҲ—жӣҙй•ҝпјҹ
    dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);

    иҮіжӯӨпјҢзҠ¶жҖҒиҪ¬з§»ж–№зЁӢе°ұеҶҷеҮәжқҘдәҶпјҢж №жҚ® dp ж•°з»„зҡ„е®ҡд№үпјҢжҲ‘д»¬иҰҒжұӮзҡ„е°ұжҳҜ dp[0][n - 1]пјҢд№ҹе°ұжҳҜж•ҙдёӘ s зҡ„жңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—зҡ„й•ҝеәҰгҖӮ

    дёүгҖҒд»Јз Ғе®һзҺ°¶

    йҰ–е…ҲжҳҺзЎ®дёҖдёӢ base caseпјҢеҰӮжһңеҸӘжңүдёҖдёӘеӯ—з¬ҰпјҢжҳҫз„¶жңҖй•ҝеӣһж–ҮеӯҗеәҸеҲ—й•ҝеәҰжҳҜ 1пјҢд№ҹе°ұжҳҜ dp[i][j] = 1 (i == j)гҖӮ

    еӣ дёә i иӮҜе®ҡе°ҸдәҺзӯүдәҺ jпјҢжүҖд»ҘеҜ№дәҺйӮЈдәӣ i > j зҡ„дҪҚзҪ®пјҢж №жң¬дёҚеӯҳеңЁд»Җд№ҲеӯҗеәҸеҲ—пјҢеә”иҜҘеҲқе§ӢеҢ–дёә 0гҖӮ

    еҸҰеӨ–пјҢзңӢзңӢеҲҡжүҚеҶҷзҡ„зҠ¶жҖҒиҪ¬з§»ж–№зЁӢпјҢжғіжұӮ dp[i][j] йңҖиҰҒзҹҘйҒ“ dp[i+1][j-1]пјҢdp[i+1][j]пјҢdp[i][j-1] иҝҷдёүдёӘдҪҚзҪ®пјӣеҶҚзңӢзңӢжҲ‘д»¬зЎ®е®ҡзҡ„ base caseпјҢеЎ«е…Ҙ dp ж•°з»„д№ӢеҗҺжҳҜиҝҷж ·пјҡ

    дёәдәҶдҝқиҜҒжҜҸж¬Ўи®Ўз®— dp[i][j]пјҢе·ҰдёӢеҸіж–№еҗ‘зҡ„дҪҚзҪ®е·Із»Ҹиў«и®Ўз®—еҮәжқҘпјҢеҸӘиғҪж–ңзқҖйҒҚеҺҶжҲ–иҖ…еҸҚзқҖйҒҚеҺҶпјҡ

    жҲ‘йҖүжӢ©еҸҚзқҖйҒҚеҺҶпјҢд»Јз ҒеҰӮдёӢпјҡ

    int longestPalindromeSubseq(string s) {
    int n = s.size();
    // dp ж•°з»„е…ЁйғЁеҲқе§ӢеҢ–дёә 0
    vector<vector<int>> dp(n, vector<int>(n, 0));
    // base case
    for (int i = 0; i < n; i++)
        dp[i][i] = 1;
    // еҸҚзқҖйҒҚеҺҶдҝқиҜҒжӯЈзЎ®зҡ„зҠ¶жҖҒиҪ¬з§»
    for (int i = n - 1; i >= 0; i--) {
        for (int j = i + 1; j < n; j++) {
            // зҠ¶жҖҒиҪ¬з§»ж–№зЁӢ
            if (s[i] == s[j])
                dp[i][j] = dp[i + 1][j - 1] + 2;
            else
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);
        }
    }
    // ж•ҙдёӘ s зҡ„жңҖй•ҝеӣһж–ҮеӯҗдёІй•ҝеәҰ
    return dp[0][n - 1];
}
